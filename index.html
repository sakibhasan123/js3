Question1.Find the number of digits

Answer.

        const Find_Digits = (N) => 
{
	 let count = 0;
	 while(N >= 1 ){
	   count++;
	   N = N / 10;
	 }
	 return count;
};

Question2.Find the Fives.

Answer.

        const Find_Five = (N) => 
{
	 let countFive = 0;
	 while(N >= 1){
	   let rem = Math.floor(N % 10);
	   if(rem === 5) countFive++;
	   N = N / 10;
	 }
	 return countFive;
};

Question3.Find Sum

Answer.

        const findSum = (n) => {
    let sum = 0;
    let i = 0;
    while(i <= n){
      if(!(i%2)) sum = sum + i;
      i++;
    }
    return sum;
};

Question4.Find the sum of the digits of a given number.

Answer.

        const Number_Sum = (N) => 
{
	let digitSum = 0;
	while(N >= 1){
	  digitSum += Math.floor(N % 10);
	  N = N / 10;
	}
	return digitSum;
};

Question6.Print the Pattern.

Answer.

        const Print_pattern = (N) => 
{
  let pattern = '';
  for(let i = 0; i < N; i++){
    for(let j = 0; j <= i; j++){
      pattern += '*';
    }
    pattern += '\n';
  }
  console.log(pattern);
};
